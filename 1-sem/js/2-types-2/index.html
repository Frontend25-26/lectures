Продвинутое объявление свойств объекта

							
								Object.defineProperty(object, propertyName, descriptor)
  • object - объект, в котором объявляется свойство
  • properyName - имя свойства, которое нужно объявить или модифицировать
  • descriptor - объект, который описывает поведение свойства

Descriptor

							
								{
									value, // значение свойства
									writable, // если `true`, можно изменить
									enumerable, // если `true`, перечисляется в циклах
									configurable, // если `true`, свойство можно удалить,
												  		  // а атрибуты дескриптора изменить
								}

Глобальное запечатывание объекта

Object.preventExtensions(object)
Запрещает добавлять новые свойства в объект 
							
								let school = {
									name: 'Хогвартс',
									founded: 990
								};

								Object.preventExtensions(school);

								school.age = 1035;
								console.log(school) // { name: 'Хогвартс', founded: 990 }

								delete school.founded
								console.log(school) // { name: 'Хогвартс' }

								school.name = 'Дурмстранг'
								console.log(school) // { name: 'Дурмстранг' }

Глобальное запечатывание объекта

Object.seal(object)
Запрещает добавлять/удалять свойства. Устанавливает configurable: false для всех существующих свойств 
							
								let school = {
									name: 'Хогвартс',
									founded: 990
								};

								Object.seal(school);

								school.age = 1035;
								console.log(school) // { name: 'Хогвартс', founded: 990 }

								delete school.founded
								console.log(school) // { name: 'Хогвартс', founded: 990 }

Глобальное запечатывание объекта

Object.freeze(object)
Запрещает добавлять/удалять/изменять свойства. Устанавливает configurable: false, writable: false для всех существующих свойств 
							
								let school = {
									name: 'Хогвартс',
									founded: 990
								};

								Object.freeze(school);

								school.age = 1035;
								console.log(school) // { name: 'Хогвартс', founded: 990 }

								delete school.founded
								console.log(school) // { name: 'Хогвартс', founded: 990 }

								school.name = 'Дурмстранг'
								console.log(school) // { name: 'Хогвартс', founded: 990 }

Глобальное запечатывание объекта

  • Object.isExtensible(object) - Возвращает false, если добавление новых свойств запрещено, иначе true
  • Object.isSealed(object) - Возвращает false, если добавление/удаление новых свойств запрещено и для всех существующих свойств установлено configurable: false
  • Object.isFrozen(object) - Возвращает false, если добавление/удаление/изменение новых свойств запрещено, и для всех текущих свойств установлено configurable: false, writable: false

Упорядочены ли свойства объекта?

  • cвойства с целочисленными ключами сортируются по возрастанию
  • остальные располагаются в порядке создания - сначала строковые, потом символьные

Что значит "целочисленные ключи"?

целочисленное свойство - строка, которая может быть преобразована в целое число и обратно без изменений 

							
								let postIndexes = {
									'119021': 'Яндекс',
									' 119021 ': 'Яндекс'
								}

'119021' - целочисленное имя свойства, потому что если его преобразовать в целое число, а затем обратно в строку, то оно не изменится, а ' 119021 ' - нет

Порядок свойств объекта. Пример 1

							
								let postIndexes = {
									'119021': 'Яндекс',
									'2':      'Школа Чародейства',
									'1':      'Хогвартс',
								}

								for (key in postIndexes) {
									console.log(key); // '1', '2', '119021'
								}

Порядок свойств объекта. Пример 2

							
								let postIndexes = {
									'test2': 'test2',
									'test1': 'test1',

									'119021': 'Яндекс',
									'2':      'Школа Чародейства',
									'1':      'Хогвартс',
								}

								for (key in postIndexes) {
									console.log(key);
									// '1', '2', '119021', 'test2', 'test1'
								}

Преобразование объектов в примитивы

Правила преобразования

  • В логическом контексте все объекты являются true
  • Числовое преобразование происходит, когда мы вычитаем объекты или применяем математические функции
  • В остальных случаях происходит преобразование к строке

Преобразование к boolean

							
								let school = {};
								Boolean(school); // true;

								let school = { name: 'Хогвартс' };
								Boolean(school); // true;

Преобразование к number и string

							
								let school = { name: 'Хогвартс' };

								Number(school); // NaN
								String(school); // '[object Object]'

Пример

							
								let student1 = {
									name: 'Harry Potter',
								}

								let student2 = {
									name: 'Draco Malfoy',
								}

								let students = {
									[student1]: "Gryffindor", // курс
									[student2]: "Slytherin", // курс
								};

								console.log(students); // ?
							
						
							
								{ '[object Object]': 'Slytherin' }

Как JavaScript понимает, какое преобразование применить?

Существует 3 варианта преобразования типов (хинты)

  • string
  • number
  • default

Хинт "string"

Для преобразования объекта к строке, когда мы выполняем операцию над объектом, которая ожидает строку

							
								let student = {
									name: 'Harry Potter',
								}

								let students = {
									[student]: "Gryffindor" // курс
								};

								console.log(students); // {[object Object]: "Gryffindor"}

Хинт "number"

Для преобразования объекта к числу, в случае математических операций

							
								// явное преобразование
								Number(school);

								// математические (не считая бинарного плюса)
								let n = +school; // унарный плюс
								let delta = date1 - date2;

								// сравнения больше/меньше
								let greater = student1 > student2;

Хинт "default"

Используется, когда оператор "не уверен", какой тип ожидать

							
								console.log(object1 + object2);

Все встроенные объекты, кроме объекта Date, реализуют default преобразование тем же способом, что и number

Как происходит преобразование

							
								function ToPrimitive(input, preferredType) {

									switch (preferredType){
									  case Number:
											return toNumber(input);
									  case String:
											return toString(input);
									  default:
											return toNumber(input);
									}

									function isPrimitive(value){
									  return value !== Object(value);
									}

									function toString(){
									  if (isPrimitive(input.toString())) return input.toString();
									  if (isPrimitive(input.valueOf())) return input.valueOf();
									  throw new TypeError();
									}

									function toNumber(){
									  if (isPrimitive(input.valueOf())) return input.valueOf();
									  if (isPrimitive(input.toString())) return input.toString();
									  throw new TypeError();
									}
								}

Преобразование Symbol.toPrimitive

							
								let school = { name: 'Хогвартс' };

								// Добавляем кастомное преобразование
								school[Symbol.toPrimitive] = function(hint) {

									// hint = "string" или "number" или "default"
									if (hint === 'string') {
										return `Школа: ${this.name}`;
									} else if (hint === 'number') {
										return 990;
									} else { // hint === 'default'
										return this.name;
									}

								};

Преобразование toString и valueOf

							
								let school = {
									name: 'Хогвартс',
									toString: function() {
										return 'Школа магии'
									},
									valueOf: function() {
										return 5;
									},
								};

								school + ' и волшебства' // Школа магии и волшебства
								school + 5 // 10

Геттеры и сеттеры

Вспомним пример

							
								let headmaster = {
									firstName: 'Альбус',
									lastName: 'Дамблдор',

									fullName: function () {
										return `${this.firstName} ${this.lastName}`
									}
								};

								headmaster.fullName(); // Альбус Дамблдор

firstName, lastName, fullName - свойства-данные

Cвойства-аксессоры

							
								let headmaster = {
									get fullName() {
										// геттер, срабатывает при чтении headmaster.fullName
									},

									set fullName(value) {
										// сеттер, срабатывает при записи headmaster.fullName = value
									}
								};

Пример get

							
								let headmaster = {
									firstName: 'Альбус',
									lastName: 'Дамблдор',

									get fullName() {
										return `${this.firstName} ${this.lastName}`;
									}
								};

								headmaster.fullName; // Альбус Дамблдор

Пример set

							
								let headmaster = {
									firstName: 'Альбус',
									lastName: 'Дамблдор',

									get fullName() {
										return `${this.firstName} ${this.lastName}`;
									},

									set fullName(value) {
										[this.firstName, this.lastName] = value.split(" ");
									}
								};

								console.log(headmaster.fullName); // "Альбус Дамблдор"

								headmaster.fullName = "Северус Снегг"; // присваиваем значение свойству

								console.log(headmaster.firstName); // "Северус"
								console.log(headmaster.lastName);  // "Снегг"

								console.log(headmaster.fullName);  // "Северус Снегг"

Дескрипторы свойств доступа

							
								let headmaster = {
									firstName: 'Альбус',
									lastName: 'Дамблдор',
								};

								Object.defineProperty(headmaster, 'fullName', {
									get() {
										return `${this.firstName} ${this.lastName}`;
									},
									set(value) {
										[this.firstName, this.lastName] = value.split(" ");
									}
								});

Дескриптор свойства-аксессора нельзя совмещать с дескриптором свойства-данных

							
								// дескриптор свойства-данных
								{
									value,
									writable,
									enumerable,
									configurable
								}

								// дескриптор свойства-аксессора
								{
									get,
									set,
									enumerable, // false
									configurable // false
								}

Сборка мусора

Главной концепцией управления памятью является принцип достижимости

«Достижимые» значения называются те значения, которые доступны или используются

Примеры базового множества достижимых значений, которые не могут быть удалены

  • Глобальные переменные
  • Выполняемая в данный момент функция, её локальные переменные и параметры
  • Функции, локальные переменные и параметры в текущей цепочке вложенных вызовов

Любое другое значение считается достижимым, если оно доступно из базового множества по ссылке или по цепочке ссылок.

Удаление единственной ссылки на объект

							
								let school = { name: 'Хогвартс' };
							
						
							
								school = null;

Удаление одной из 2 ссылок на объект

							
								let school_1 = { name: 'Хогвартс' };
								let school_2 = school_1;
							
						
							
								school_1 = null;

Пример перекрестной связи

							
								function makeFriends() {
									let harry = { name: 'Гарри' };
									let ron = { name: 'Рон' };

									harry.friend = ron;
									ron.friend = harry;

									return {
										harry,
										ron
									}
								}

								let friends = makeFriends();

Пример перекрестной связи

Пример перекрестной связи

							
								delete friends.harry
								delete friends.ron.friend

Пример перекрестной связи

							
								friends = null;

Основной алгоритм сбора мусора

«алгоритм пометок»

  1. Определение корней — сборщик мусора находит корневые объекты и запоминает их
  2. Обход по ссылкам — он проходит по всем ссылкам, исходящим от корней, и помечает все объекты, до которых можно добраться
  3. Рекурсивный обход — для каждого найденного объекта сборщик также просматривает его ссылки и помечает доступные объекты
  4. Результат обхода — все объекты, до которых можно добраться, считаются живыми
  5. Удаление мусора — все объекты, которые не были помечены, считаются недостижимыми и удаляются из памяти

Оптимизации сборки мусора


Оптимизация Что делает Зачем нужна
Generational Делит память на «молодые» и «старые» объекты Чтобы чаще чистить короткоживущие
Incremental Делит сборку на шаги Чтобы не тормозить приложение
Idle-time Запускает GC, когда движок свободен Чтобы не мешать работе кода

Итого

  • Сборка мусора выполняется автоматически
  • Объекты сохраняются в памяти, пока они достижимы
  • Если на объект есть ссылка не означает, что объект достижим

Коллекция Map

Коллекция Map

Коллекция ключ/значение, как и Object

							
								let map = new Map() // создание коллекции
								map.set(key, value) // добавление записи
								map.get(key) // получение записи
								map.has(key) // проверка наличия записи
								map.delete(key) // удаление записи
								map.clear() // очистка коллекции
								map.size // кол-во записей

Пример коллекции Map

							
								let students = new Map();

								students.set('Гарри', 'Гриффиндор');
								students.set('Рон', 'Гриффиндор');
								students.set('Гермиона', 'Гриффиндор');

								students.size // 3;

								students.get('Гарри'); // Гриффиндор

								students.has('Гарри'); // true

								students.delete('Гарри'); // true
								students.has('Гарри'); // false
								students.size // 2;

Пример коллекции Map

							
								let students = new Map();

								students.set('Гарри', 'Гриффиндор');
								students.set('Рон', 'Гриффиндор');
								students.set('Гермиона', 'Гриффиндор');
							
						
							
								let students = new Map([
									['Гарри', 'Гриффиндор'],
									['Рон', 'Гриффиндор'],
									['Гермиона', 'Гриффиндор'],
								]);

Map похож на объект, для чего использовать Map?

  • Можно использовать любые типы данных для ключей
  • Можно узнать кол-во элементов коллекции без дополнительных затрат
  • Перебор коллекции происходит в том же порядке, в каком происходило добавление элементов

Map VS Object. Пример с объектом

							
								let skippedLectures = {};

								let student1 = { name: 'Гарри' }
								let student2 = { name: 'Рон' }

								skippedLectures[student1] = 3;
								skippedLectures[student2] = 4;
							
						
							
								console.log(skippedLectures);
								// {'[object Object]': 4}
							
						
							
								// Сколько записей в объекте?
								Object.keys(skippedLectures).length; // 1

Map VS Object. Пример с Map

							
								let skippedLectures = new Map();

								let student1 = { name: 'Гарри' }
								let student2 = { name: 'Рон' }

								skippedLectures.set(student1, 3);
								skippedLectures.set(student2, 4);
							
						
							
								skippedLectures.get(student1); // 3
								skippedLectures.get(student2); // 4
							
						
							
								// Сколько записей в коллекции?
								skippedLectures.size; // 2

Встроенные методы для перебора коллекции Map

  • map.keys() - возвращает итерируемый объект по ключам
  • map.values() - возвращает итерируемый объект по значениям
  • map.entries() - возвращает итерируемый объект по парам вида [ключ, значение]

Перебор коллекции Map

							
								let skippedLectures = new Map([
									["Гарри", 3],
									["Рон", 4],
									["Гермиона", 0]
								]);

								// перебор по ключам
								for (let name of skippedLectures.keys()) {
									console.log(name); // Гарри, Рон, Гермиона
								}

								// перебор по значениям)
								for (let amount of skippedLectures.values()) {
									console.log(amount); // 3, 4, 0
								}

								// перебор по элементам в формате [ключ, значение]
								for (let entry of skippedLectures.entries()) {
									console.log(entry);
									// ["Гарри", 3]
									// ["Рон", 4]
									// ["Гермиона", 0]
								}

								// сокращенная запись предыдущего примера
								for (let entry of skippedLectures) {
								    console.log(entry);
									// ["Гарри", 3]
									// ["Рон", 4]
									// ["Гермиона", 0]
								}

								// выполняем функцию для каждой пары (ключ, значение)
								skippedLectures.forEach((value, key, map) => {
								  console.log(`${key}: ${value}`);
									// 'Гарри: 3'
									// 'Рон: 4'
									// 'Гермиона: 4'
								});

Из Object в Map

							
								let school = {
									name: 'Хогвартс',
									founded: 990
								};

								Object.entries(school);
								// [
								//    ['name', 'Хогвартс']
								//    ['founded', 990]
								// ]

								let map = new Map(Object.entries(school));
								map.get('name'); // 'Хогвартс'

Наоборот, из Map в Object

							
								let school = new Map([
									['name', 'Хогвартс'],
									['founded', 990]
								]);

								Object.fromEntries([
									['name', 'Хогвартс'],
									['founded', 990]
								]);
								// { name: 'Хогвартс', founded: 990 }

								let obj = Object.fromEntries(school);
								obj.name; // 'Хогвартс'

Коллекция Set

Коллекции вида «множество» значений (без ключей), где каждое значение может появляться только один раз

							
								new Set(iterable) // создание коллекции

								set.has(value) // проверка наличия записи
								set.delete(value) // удаление записи
								set.clear() // очистка коллекции
								set.size // кол-во записей

Пример коллекции Set

							
								let set = new Set();

								set.add(100);
								set.add(150);
								set.add(200);

								set.size; // 3

								set.has(200); // true

Пример коллекции Set

							
								let set = new Set();

								set.add(100);
								set.add(150);
								set.add(200);
							
						
							
								let set = new Set([100,150,200]);

Особенности Set

- игнорируются дубликаты

							
								let set = new Set();

								set.add(1);
								set.add(2);
								set.add(2);

								set.size; // 2

Встроенные методы для перебора коллекции Set

  • set.keys() - возвращает перебираемый объект для значений
  • set.values() - то же самое, что и set.keys() (для совместимости с Map)
  • set.entries() - возвращает перебираемый объект для пар вида [значение, значение] (для совместимости с Map)

Перебор коллекции Set

							
								let set = new Set([1,2,3]);

								for (let value of set) {
									console.log(value); // 1, 2, 3
								}

								set.forEach((value, valueCopy, set) => {
									console.log(value); // 1, 2, 3
								})

								for (let value of set.keys()) {
									console.log(value); // 1, 2, 3
								}

								for (let value of set.values()) {
									console.log(value); // 1, 2, 3
								}

								for (let value of set.entries()) {
									console.log(value);
									// [1, 1]
									// [2, 2]
									// [3, 3]
								}

								Array.from(set); // [1, 2, 3]

Коллекция WeakMap/WeakSet

Вспоминаем концепцию сборщика мусора

							
								// school - ссылка на объект
								let school = { name: 'Хогвартс' };

								// Делаем объект недостижимым
								school = null;

								// Сборщик мусора удалит объект
							
						
							
								// school - ссылка на объект
								let school = { name: 'Хогвартс' };

								let schools = new Map();
								schools.set(school, true);

								// Делаем объект недостижимым
								school = null;

								// Сборщик мусора НЕ удалит объект

Отличия WeakMap от Map

  • В качестве ключа можно использовать только объекты
  • Коллекция не предотвращает удаление объектов, которые выступают в качестве ключей коллекции
  • Не поддерживает перебор и методы keys(),values() и entries()
  • Отсутствует свойство size

Рассмотрим пример WeakMap

							
								let person = { name: "Гарри" };

								let students = new WeakMap();
								students.set(person, true);

								person = null;

								// students станет пустой коллекцией
								// когда отработает сборщик мусора

Рассмотрим пример WeakMap

							
								const privateData = new WeakMap();

								function createUser(user, id) {
									privateData.set(user, { id });
									return user;
								}

								function getId(user) {
									return privateData.get(user).id;
								}

Отличия WeakSet от Set

(аналогично WeakMap)

  • Можем добавлять только объекты
  • Коллекция не предотвращает удаление объектов, которые выступают в качестве значений коллекции
  • Не поддерживает перебор и методы keys(),values() и entries()
  • Отсутствует свойство size

Рассмотрим пример WeakSet

							
								let person = { name: "Гарри" };

								let whoAttendedTheLecture = new WeakSet();
								whoAttendedTheLecture.add(person);

								whoAttendedTheLecture.has(person); // true

								person = null;
								// whoAttendedTheLecture станет пустой коллекцией
								// когда отработает сборщик мусора
